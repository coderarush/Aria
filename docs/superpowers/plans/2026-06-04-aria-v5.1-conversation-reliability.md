# Aria v5.1 — Conversation Reliability Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make talking to Aria premium — full talk-over barge-in via a real echo canceller, plus a wake-always / never-deaf / faster / less-mis-hearing audio pipeline, all 100% local and free.

**Architecture:** One `AVAudioEngine` (`AudioBus`) owns both the mic and Aria's TTS playback, so we have her exact audio as the echo-cancel reference. A vendored C echo canceller (`speexdsp`) removes her voice from the mic; the cleaned stream feeds the recognizer (fed manually via `append`) and a VAD. Because the recognizer never hears Aria, talk-over barge-in works and self-trigger is gone; owning one engine kills the deaf/wake bugs.

**Tech Stack:** Swift 5.9 / SwiftPM, AVFoundation (AVAudioEngine, AVAudioConverter, AVAudioPlayerNode), Speech (SFSpeechRecognizer), a vendored C `speexdsp` echo canceller (BSD), XCTest.

**Spec:** `docs/superpowers/specs/2026-06-04-aria-v5.1-conversation-reliability-design.md`

**Branch:** `aria-v5.1` (off `main` @ v4.0.0). Baseline: 100 tests, `swift build` clean.

    private let frame = 160, rate = 16000

    /// A near-end that is PURE echo (a scaled, delayed far-end) should be largely

    func testCancelsEchoOnlyNearEnd() {

        var farHist = [Int16](repeating: 0, count: frame)
        var echoEnergy = 0.0, residualEnergy = 0.0

        for i in 0..<200 {
            // Far-end: white-ish noise.
            let far = (0..<frame).map { _ in Int16(Int.random(in: -8000...8000, using: &rng)) }

import Foundation
import CSpeexDSP

/// Swift wrapper over speexdsp's MDF acoustic echo canceller. Feed it the near-end
/// (mic) and the far-end (what we're playing — Aria's TTS) in matched frames; get
/// back the near-end with the echo removed. Not thread-safe; the owner (AudioBus)
/// calls it from one audio queue.
final class EchoCanceller {
    private let frameSize: Int
    private var state: OpaquePointer?      // SpeexEchoState*
    private var preprocess: OpaquePointer? // SpeexPreprocessState* (residual suppression)

    init(frameSize: Int = 160, filterTaps: Int = 160 * 16, sampleRate: Int = 16000) {
        self.frameSize = frameSize
        state = speex_echo_state_init(Int32(frameSize), Int32(filterTaps))
        var rate = Int32(sampleRate)
        speex_echo_ctl(state, SPEEX_ECHO_SET_SAMPLING_RATE, &rate)
        preprocess = speex_preprocess_state_init(Int32(frameSize), Int32(sampleRate))
        // Let the preprocessor use the echo state for residual echo suppression.
        speex_preprocess_ctl(preprocess, SPEEX_PREPROCESS_SET_ECHO_STATE, state)
    }

    deinit {
        if let s = state { speex_echo_state_destroy(s) }
        if let p = preprocess { speex_preprocess_state_destroy(p) }
    }

    /// near.count and far.count must equal frameSize.
    func process(near: [Int16], far: [Int16]) -> [Int16] {
        guard near.count == frameSize, far.count == frameSize else { return near }
        var out = [Int16](repeating: 0, count: frameSize)
        near.withUnsafeBufferPointer { n in
            far.withUnsafeBufferPointer { f in
                out.withUnsafeMutableBufferPointer { o in
                    speex_echo_cancellation(state, n.baseAddress, f.baseAddress, o.baseAddress)
                }
            }
        }
        out.withUnsafeMutableBufferPointer { o in
            _ = speex_preprocess_run(preprocess, o.baseAddress)  // residual echo + noise
        }
        return out
    }
}
```
If the speex `ctl` constants (`SPEEX_ECHO_SET_SAMPLING_RATE`, etc.) don't import as Swift Int32, pass the literal ints from `speex_echo.h`/`speex_preprocess.h` directly.

import Foundation

enum AudioFrameMath {
    static func int16(fromFloat f: Float) -> Int16 {
        let scaled = max(-1, min(1, f)) * 32767
        return Int16(scaled.rounded())
    }
    static func int16s(fromFloats f: [Float]) -> [Int16] { f.map(int16(fromFloat:)) }
}

/// Accumulates Int16 samples and hands them out in fixed-size frames, buffering the
/// remainder. Used to turn variable-size audio taps into the AEC's fixed frames.
struct FrameRing {
    let frameSize: Int
    private var buf: [Int16] = []
    init(frameSize: Int) { self.frameSize = frameSize }
    mutating func push(_ samples: [Int16]) { buf.append(contentsOf: samples) }
    mutating func pop() -> [Int16]? {
        guard buf.count >= frameSize else { return nil }
        let frame = Array(buf.prefix(frameSize))
        buf.removeFirst(frameSize)
        return frame
    }
}
```

```swift
import Foundation
import AVFoundation

/// The single owner of audio I/O. One AVAudioEngine taps the mic and plays Aria's
/// TTS through a player node, so we have her exact audio as the AEC far-end. It
/// resamples mic + TTS to 16 kHz mono, runs the EchoCanceller, and publishes the
/// cleaned near-end as Int16 frames via `onCleanedFrame`. Free + local.
final class AudioBus {
    static let aecRate: Double = 16000
    static let frameSize = 160                 // 10 ms @ 16 kHz

    private let engine = AVAudioEngine()
    private let player = AVAudioPlayerNode()
    private let aec = EchoCanceller(frameSize: frameSize, filterTaps: frameSize * 16, sampleRate: Int(aecRate))
    private let aecFormat = AVAudioFormat(commonFormat: .pcmFormatInt16, sampleRate: aecRate, channels: 1, interleaved: true)!

    private var micRing = FrameRing(frameSize: frameSize)
    private var farRing = FrameRing(frameSize: frameSize)
    private var micConverter: AVAudioConverter?
    private var farConverter: AVAudioConverter?

    /// Emits cleaned 16 kHz mono Int16 frames (echo removed).
    var onCleanedFrame: (([Int16]) -> Void)?
    /// True while Aria's TTS is actively playing (drives barge-in).
    private(set) var isPlaying = false

    func start() throws {
        let input = engine.inputNode
        let inFormat = input.inputFormat(forBus: 0)
        micConverter = AVAudioConverter(from: inFormat, to: aecFormat)

        engine.attach(player)
        engine.connect(player, to: engine.mainMixerNode, format: nil)
        // Keep Aria's playback OUT of the mic's near-end? No — we WANT it acoustically
        // (that's the echo we cancel). The far-end reference comes from the PCM we
        // schedule (see playReference), captured into farRing.

        input.installTap(onBus: 0, bufferSize: 1024, format: inFormat) { [weak self] buf, _ in
            self?.handleMic(buf)
        }
        engine.prepare()
        try engine.start()
    }

    func stop() {
        engine.inputNode.removeTap(onBus: 0)
        engine.stop()
    }

    private func handleMic(_ buffer: AVAudioPCMBuffer) {
        guard let conv = micConverter,
              let i16 = AudioBus.convert(buffer, with: conv, to: aecFormat) else { return }
        micRing.push(i16)
        drain()
    }

    /// Pair mic + far frames and run AEC. If we have no far audio queued (Aria
    /// silent), use a zero far-end (AEC becomes a near-passthrough).
    private func drain() {
        while let near = micRing.pop() {
            let far = farRing.pop() ?? [Int16](repeating: 0, count: AudioBus.frameSize)
            let cleaned = aec.process(near: near, far: far)
            onCleanedFrame?(cleaned)
        }
    }

    /// Play Aria's TTS (16-bit PCM at `pcmRate`) AND register it as the AEC far-end.
    func playReference(pcm: Data, pcmRate: Double, onDone: @escaping () -> Void) {
        // 1) Register as far-end (resample to 16k Int16 into farRing).
        if let i16 = AudioBus.resampleInt16(pcm: pcm, fromRate: pcmRate, toFormat: aecFormat) {
            farRing.push(i16)
        }
        // 2) Schedule playback through the player node at its native rate.
        guard let srcFormat = AVAudioFormat(commonFormat: .pcmFormatInt16, sampleRate: pcmRate, channels: 1, interleaved: true),
              let buf = AudioBus.pcmBuffer(from: pcm, format: srcFormat) else { onDone(); return }
        isPlaying = true
        if !player.isPlaying { player.play() }
        player.scheduleBuffer(buf, at: nil) { [weak self] in
            DispatchQueue.main.async { self?.isPlaying = false; onDone() }
        }
    }

    func stopPlayback() {
        player.stop()
        farRing = FrameRing(frameSize: AudioBus.frameSize)  // drop pending reference
        isPlaying = false
    }

    // MARK: conversion helpers (thin AVAudioConverter wrappers)
    private static func convert(_ buffer: AVAudioPCMBuffer, with conv: AVAudioConverter, to format: AVAudioFormat) -> [Int16]? {
        let ratio = format.sampleRate / buffer.format.sampleRate
        let cap = AVAudioFrameCount(Double(buffer.frameLength) * ratio + 16)
        guard let out = AVAudioPCMBuffer(pcmFormat: format, frameCapacity: cap) else { return nil }
        var fed = false
        var err: NSError?
        conv.convert(to: out, error: &err) { _, status in
            if fed { status.pointee = .noDataNow; return nil }
            fed = true; status.pointee = .haveData; return buffer
        }
        if err != nil { return nil }
        guard let ch = out.int16ChannelData else { return nil }
        return Array(UnsafeBufferPointer(start: ch[0], count: Int(out.frameLength)))
    }
    private static func resampleInt16(pcm: Data, fromRate: Double, toFormat: AVAudioFormat) -> [Int16]? {
        guard let src = AVAudioFormat(commonFormat: .pcmFormatInt16, sampleRate: fromRate, channels: 1, interleaved: true),
              let buf = pcmBuffer(from: pcm, format: src),
              let conv = AVAudioConverter(from: src, to: toFormat) else { return nil }
        return convert(buf, with: conv, to: toFormat)
    }
    private static func pcmBuffer(from pcm: Data, format: AVAudioFormat) -> AVAudioPCMBuffer? {
        let frames = AVAudioFrameCount(pcm.count / 2)
        guard frames > 0, let buf = AVAudioPCMBuffer(pcmFormat: format, frameCapacity: frames) else { return nil }
        buf.frameLength = frames
        pcm.withUnsafeBytes { raw in
            if let base = buf.int16ChannelData?[0], let src = raw.bindMemory(to: Int16.self).baseAddress {
                base.update(from: src, count: Int(frames))
            }
        }
        return buf
    }
}
```

Create `Sources/Aria/Core/Audio/RecognitionLifecycle.swift`:
```swift
import Foundation

/// Pure decision logic for keeping recognition alive: which session is current
/// (so a superseded task's completion can't trigger a restart) and a watchdog that
/// detects a silently-dead recognizer (no audio flowing).
struct RecognitionLifecycle {
    private var current = 0
    private var lastAudioAt: TimeInterval = 0

    mutating func begin() -> Int { current += 1; return current }
    func shouldRestart(forSession id: Int) -> Bool { id == current }
    mutating func sawAudio(at t: TimeInterval) { lastAudioAt = t }
    func watchdogExpired(now: TimeInterval, timeout: TimeInterval) -> Bool {
        now - lastAudioAt > timeout

import Foundation

/// Detects the user talking OVER Aria. Runs on the cleaned near-end (Aria's voice
/// already removed by the AEC), so sustained energy = the user. Only active while
/// Aria is speaking; fires `onBarge` once per onset.
final class BargeController {
    private let onsetFrames: Int
    private let energyThreshold: Double
    private var consecutive = 0
    private var playing = false
    var onBarge: (() -> Void)?

    init(onsetFrames: Int = 4, energyThreshold: Double = 600) {
        self.onsetFrames = onsetFrames
        self.energyThreshold = energyThreshold
    }

    func setPlaying(_ v: Bool) { playing = v; if !v { consecutive = 0 } }

    func feed(_ frame: [Int16]) {
        guard playing else { return }
        let rms = sqrt(frame.reduce(0.0) { $0 + Double($1) * Double($1) } / Double(max(frame.count, 1)))
        if rms >= energyThreshold {
            consecutive += 1
            if consecutive == onsetFrames { onBarge?() }
        } else {
            consecutive = 0
        }
    }
}
```
